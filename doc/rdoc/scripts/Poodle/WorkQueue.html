def add(uri, referer, checksum)
    @items.synchronize do
        id = unique_id(uri)
        unless @crawled.include?(id)
            @items << [uri, referer, checksum]
            @crawled.add(id)
        end
    end
end

def remove()
    yielded = false
    item = nil
    @items.synchronize do
        item = @items.shift
        if @items.empty?
            @processed << (yield item) if (block_given? and item)
            yielded = true
        end
    end
    if (block_given? and not yielded)
        @processed << (yield item)
    end
end

def unique_id(uri)
    digest = Digest::MD5.new().update(uri.normalize().to_s)
    digest.hexdigest
end
